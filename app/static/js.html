    <div id="name-group" class="input-group"></div>

    <div id="email-group" class="input-group">
        <label for="exampleInputEmail1">Email address</label>
        <input type="email" class="form-control"  required name="email_add" id="exampleInputEmail1" placeholder="Enter email" >
    </div>

    <br>

    <div id="submit-group" class="input-group">
        <button type="submit" class="btn btn-large">Submit</button>
    </div>

<div id="success-msg"></div>

JavaScript Code

                            
<script type="text/javascript">
$(function() {
//twitter bootstrap script
    $('#exampleForm').validate({

        rules: {
            email_add: {
                required: true,
                email: true
            }
        },
        messages: {
            email_add: "Please enter a valid email address NOW"
        },

        submitHandler: function(form) {

            // Console Output to see input when submit handler is called
            console.log($(form).attr('action'));
            console.log($(form).serialize());


            // Check and remove errors if preset

            if ($('#email-group').hasClass('has-error')) {
                // remove the error class
                $('#email-group').removeClass('has-error');

            }

            if ($('p').hasClass('alert-danger')) {

                $('#message_error').remove();
            } else {
                alert('class not found');
            }

        $.post( $(form).attr('action'), $(form).serialize() , function(data) {
            console.log(data);
            console.log(data.success);


            var response = $.parseJSON(data);
            if (response.success){
                // alert ('success');

                $(form)[0].reset();
                $(form).hide();  // Hide the Form
                $('#success-msg').append('<p class="alert alert-info text-center">' + response.message + '</p>');
            } else {
                if (response.errors.email) {
                    // add the error class to show red input
                    $('#email-group').addClass('has-error');

                    // add the actual error message under our input
                    $('#email-group').append('<span class="help-block"><small class="bg-danger">' + response.errors.email + '</small></span>');

                } else if (response.errors.message) {

                     // script was unable to send email message. let the user know
                    $('#success-msg').append('<p id="message_error" class="alert alert-danger text-center">' + response.errors.message + '</p>');
                }
            }
        });  // Post Function Close
    }   //Submit Handler Close
    });  // Validate Function Close
});// Close function
</script>
